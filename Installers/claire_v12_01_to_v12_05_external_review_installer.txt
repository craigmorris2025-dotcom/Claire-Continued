#!/usr/bin/env python3
from __future__ import annotations

import argparse
import json
import shutil
from datetime import datetime, timezone
from pathlib import Path
from typing import Any, Dict, List

INSTALLER_NAME = 'Claire v12.01-v12.05 External Review Installer'
INSTALLER_VERSION = '12.01-12.05'
DEFAULT_TARGET = Path.home() / "OneDrive" / "Desktop" / "Claire"

FOLDERS = [
  "src/claire/external_review_readiness",
  "tests/external_review_readiness",
  "data/external_review_readiness/reviewer_packets",
  "data/external_review_readiness/redaction_checks",
  "data/external_review_readiness/pilot_handoff",
  "data/external_review_readiness/reviewer_feedback",
  "docs/external_review_readiness",
  ".claire_install/reports"
]
FILES = {
  "src/claire/external_review_readiness/__init__.py": "\"\"\"Claire external review readiness.\"\"\"\n",
  "src/claire/external_review_readiness/reviewer_packet_builder.py": "from __future__ import annotations\nfrom typing import Dict, Any, List\n\nclass ReviewerPacketBuilder:\n    def build_packet(self, reviewer_type: str, artifacts: List[str]) -> Dict[str, Any]:\n        return {\"record_type\": \"reviewer_packet\", \"reviewer_type\": reviewer_type, \"artifacts\": artifacts, \"artifact_count\": len(artifacts)}\n",
  "src/claire/external_review_readiness/redaction_checklist.py": "from __future__ import annotations\nfrom typing import Dict, Any\n\nclass RedactionChecklist:\n    REQUIRED = [\"secrets_removed\", \"private_keys_removed\", \"personal_files_removed\", \"license_notes_present\"]\n    def evaluate(self, checks: Dict[str, bool]) -> Dict[str, Any]:\n        missing = [x for x in self.REQUIRED if not checks.get(x, False)]\n        return {\"status\": \"clear\" if not missing else \"blocked\", \"missing\": missing}\n",
  "src/claire/external_review_readiness/pilot_handoff_manifest.py": "from __future__ import annotations\nfrom typing import Dict, Any, List\n\nclass PilotHandoffManifest:\n    def build_manifest(self, scope: str, limitations: List[str]) -> Dict[str, Any]:\n        return {\"record_type\": \"pilot_handoff_manifest\", \"scope\": scope, \"limitations\": limitations}\n",
  "docs/external_review_readiness/v12_01_to_v12_05_external_review_plan.md": "# v12.01-v12.05 External Review / Pilot Readiness\n\nReviewer packet, redaction checklist, pilot handoff, and feedback intake.\n",
  "tests/external_review_readiness/test_external_review_readiness.py": "from claire.external_review_readiness.redaction_checklist import RedactionChecklist\nfrom claire.external_review_readiness.reviewer_packet_builder import ReviewerPacketBuilder\n\ndef test_redaction_blocks_empty():\n    assert RedactionChecklist().evaluate({})[\"status\"] == \"blocked\"\n\ndef test_packet_count():\n    assert ReviewerPacketBuilder().build_packet(\"technical\", [\"readme\"])[\"artifact_count\"] == 1\n"
}
VERSION_FILE = 'version_12_05_external_review_readiness.json'
LOCK_PAYLOAD = {
  "version": "12.05",
  "state": "external_review_ready",
  "phase": "external_review_pilot_readiness",
  "status": "installed_not_sent",
  "safety_contract": {
    "does_not_modify_runtime": True,
    "does_not_modify_routing": True,
    "does_not_modify_scoring": True,
    "does_not_modify_core_run_output": True
  }
}

def stamp() -> str:
    return datetime.now(timezone.utc).strftime("%Y%m%d_%H%M%S")

def now_iso() -> str:
    return datetime.now(timezone.utc).isoformat()

def ensure_dir(path: Path, install: bool, actions: List[Dict[str, Any]]) -> None:
    existed = path.exists()
    if install:
        path.mkdir(parents=True, exist_ok=True)
    actions.append({"action": "ensure_dir", "path": str(path), "installed": install, "existed": existed})

def write_file(path: Path, content: str, install: bool, actions: List[Dict[str, Any]]) -> None:
    existed = path.exists()
    backup = None
    if install:
        path.parent.mkdir(parents=True, exist_ok=True)
        if existed:
            backup_path = path.with_name(path.name + ".bak_" + stamp())
            shutil.copy2(path, backup_path)
            backup = str(backup_path)
        path.write_text(content, encoding="utf-8", newline="\n")
    actions.append({"action": "write_file", "path": str(path), "installed": install, "existed": existed, "backup": backup})

def main() -> int:
    parser = argparse.ArgumentParser(description=INSTALLER_NAME)
    parser.add_argument("--target", default=str(DEFAULT_TARGET), help="Claire project root")
    parser.add_argument("--install", action="store_true", help="Apply install. Without this, dry run only.")
    args = parser.parse_args()

    root = Path(args.target).expanduser().resolve()
    install = bool(args.install)

    print(f"{INSTALLER_NAME} {INSTALLER_VERSION}")
    print(f"Target: {root}")
    print("Mode:", "INSTALL" if install else "DRY RUN")

    if not root.exists():
        print(f"[ERROR] Target does not exist: {root}")
        return 2
    if not (root / "src").exists():
        print("[ERROR] This does not look like Claire root. Missing src folder.")
        return 3

    actions: List[Dict[str, Any]] = []
    for folder in FOLDERS:
        ensure_dir(root / folder, install, actions)
    for rel, content in FILES.items():
        write_file(root / rel, content, install, actions)

    payload = dict(LOCK_PAYLOAD)
    payload["installed_at_utc"] = now_iso()
    write_file(root / VERSION_FILE, json.dumps(payload, indent=2, sort_keys=True), install, actions)

    report = {
        "installer": INSTALLER_NAME,
        "installer_version": INSTALLER_VERSION,
        "installed": install,
        "installed_at_utc": now_iso(),
        "target": str(root),
        "actions": actions,
        "version_file": VERSION_FILE,
        "safety_contract": payload.get("safety_contract", {}),
        "post_install_checks": [
            "Run the targeted pytest folder listed for this installer.",
            "Confirm new src/claire package folder exists.",
            "Confirm new data and docs folders exist.",
            "Commit after tests pass."
        ]
    }

    if install:
        report_path = root / ".claire_install" / "reports" / (VERSION_FILE.replace(".json", "") + "_" + stamp() + ".json")
        report_path.parent.mkdir(parents=True, exist_ok=True)
        report_path.write_text(json.dumps(report, indent=2, sort_keys=True), encoding="utf-8")
        print(f"[OK] Report written: {report_path}")
    else:
        print("[DRY RUN] No files changed. Re-run with --install to apply.")

    print(json.dumps({
        "installed": install,
        "target": str(root),
        "installer_version": INSTALLER_VERSION,
        "folders": len(FOLDERS),
        "files": len(FILES) + 1
    }, indent=2))
    return 0

if __name__ == "__main__":
    raise SystemExit(main())
