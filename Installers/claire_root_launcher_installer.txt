#!/usr/bin/env python3
from __future__ import annotations

import argparse
import json
import shutil
from datetime import datetime, timezone
from pathlib import Path

INSTALLER_NAME = "Claire Root Launcher Installer"
INSTALLER_VERSION = "11.41"
DEFAULT_TARGET = Path.home() / "OneDrive" / "Desktop" / "Claire"

LAUNCHER_CONTENT = '@echo off\nsetlocal EnableExtensions\n\nREM ============================================================\nREM Claire Root Launcher\nREM File: LAUNCH_CLAIRE.bat\nREM Purpose:\nREM   One clean launcher from the Claire project root.\nREM   Safe replacement for older START_CLAIRE*.bat launcher clutter.\nREM ============================================================\n\ncd /d "%~dp0"\n\nset "CLAIRE_ROOT=%~dp0"\nset "CLAIRE_APP_SHELL=1"\nset "CLAIRE_LAUNCH_MODE=desktop_root"\nset "CLAIRE_DASHBOARD_PATH=src\\frontend\\command_center\\modern\\index.html"\nset "CLAIRE_LOG_DIR=logs"\n\nif not exist "%CLAIRE_LOG_DIR%" mkdir "%CLAIRE_LOG_DIR%"\n\necho.\necho ============================================================\necho Claire Syntalion - Root Launcher\necho Root: %CLAIRE_ROOT%\necho ============================================================\necho.\n\nif exist ".venv\\Scripts\\python.exe" (\n    set "CLAIRE_PYTHON=.venv\\Scripts\\python.exe"\n) else (\n    set "CLAIRE_PYTHON=python"\n)\n\necho Using Python: %CLAIRE_PYTHON%\necho.\n\nREM Optional version/dashboard sync helper.\nif exist "tools\\desktop_version_sync.py" (\n    echo Syncing desktop/version state...\n    "%CLAIRE_PYTHON%" "tools\\desktop_version_sync.py" --root "%CLAIRE_ROOT%" --mode desktop_root > "%CLAIRE_LOG_DIR%\\desktop_version_sync.out.log" 2> "%CLAIRE_LOG_DIR%\\desktop_version_sync.err.log"\n)\n\nREM Prefer the known working portable launcher route.\nif exist "tools\\portable_launcher.py" (\n    echo Starting Claire through tools\\portable_launcher.py --live --app\n    echo.\n    start "" "%CLAIRE_DASHBOARD_PATH%"\n    "%CLAIRE_PYTHON%" "tools\\portable_launcher.py" --live --app\n    goto :done\n)\n\nREM Fallback to main.py if portable launcher is unavailable.\nif exist "main.py" (\n    echo Starting Claire through main.py fallback...\n    echo.\n    start "" "%CLAIRE_DASHBOARD_PATH%"\n    "%CLAIRE_PYTHON%" "main.py"\n    goto :done\n)\n\necho.\necho [Claire Launcher Error]\necho Could not find:\necho   tools\\portable_launcher.py\necho   main.py\necho.\necho This launcher must be inside the Claire project root:\necho   C:\\Users\\Craig\\OneDrive\\Desktop\\Claire\necho.\npause\n\n:done\necho.\necho Claire launcher closed.\nendlocal\n'

def stamp():
    return datetime.now(timezone.utc).strftime("%Y%m%d_%H%M%S")

def main():
    p = argparse.ArgumentParser(description=INSTALLER_NAME)
    p.add_argument("--target", default=str(DEFAULT_TARGET))
    p.add_argument("--install", action="store_true")
    args = p.parse_args()

    root = Path(args.target).expanduser().resolve()
    install = bool(args.install)

    print(f"{INSTALLER_NAME} {INSTALLER_VERSION}")
    print(f"Target: {root}")
    print("Mode:", "INSTALL" if install else "DRY RUN")

    if not root.exists():
        print(f"[ERROR] Target does not exist: {root}")
        return 2

    launcher_path = root / "LAUNCH_CLAIRE.bat"
    actions = []

    if install:
        backup = None
        if launcher_path.exists():
            backup_path = launcher_path.with_name(launcher_path.name + ".bak_" + stamp())
            shutil.copy2(launcher_path, backup_path)
            backup = str(backup_path)

        launcher_path.write_text(LAUNCHER_CONTENT, encoding="utf-8", newline="\r\n")
        actions.append({
            "action": "write_root_launcher",
            "path": str(launcher_path),
            "backup": backup,
            "installed": True
        })

        report_dir = root / ".claire_install" / "reports"
        report_dir.mkdir(parents=True, exist_ok=True)
        report = {
            "installer": INSTALLER_NAME,
            "installer_version": INSTALLER_VERSION,
            "installed_at_utc": datetime.now(timezone.utc).isoformat(),
            "root": str(root),
            "launcher": "LAUNCH_CLAIRE.bat",
            "actions": actions,
            "safe_to_delete_old_launchers_after_confirming": [
                "START_CLAIRE.bat",
                "START_CLAIRE_DESKTOP.bat",
                "START_CLAIRE_LIVE.bat",
                "START_CLAIRE_LIVE_SAFE.bat",
                "START_CLAIRE_PORTABLE.bat",
                "START_CLAIRE_PORTABLE_SAFE.bat"
            ],
            "safety_contract": {
                "does_not_modify_runtime": True,
                "does_not_modify_scoring": True,
                "does_not_modify_routing": True,
                "root_accessible": True
            }
        }
        report_path = report_dir / ("root_launcher_install_" + stamp() + ".json")
        report_path.write_text(json.dumps(report, indent=2), encoding="utf-8")
        print(f"[OK] Created: {launcher_path}")
        print(f"[OK] Report: {report_path}")
    else:
        print("[DRY RUN] No files changed. Re-run with --install to create LAUNCH_CLAIRE.bat.")
        print(f"Would create: {launcher_path}")

    print()
    print("After install, launch Claire with:")
    print("  LAUNCH_CLAIRE.bat")
    return 0

if __name__ == "__main__":
    raise SystemExit(main())
