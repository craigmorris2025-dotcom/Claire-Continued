#!/usr/bin/env python3
from __future__ import annotations

import argparse
import json
import shutil
from datetime import datetime, timezone
from pathlib import Path
from typing import Any, Dict, List

INSTALLER_NAME = 'Claire v13 Enterprise Orchestration Installer'
INSTALLER_VERSION = '13.0-13.5'
DEFAULT_TARGET = Path.home() / "OneDrive" / "Desktop" / "Claire"

FOLDERS = [
  "src/claire/enterprise_orchestration",
  "tests/enterprise_orchestration",
  "data/enterprise_orchestration/operator_roles",
  "data/enterprise_orchestration/workspaces",
  "data/enterprise_orchestration/tenant_boundaries",
  "data/enterprise_orchestration/access_reviews",
  "docs/enterprise_orchestration",
  ".claire_install/reports"
]
FILES = {
  "src/claire/enterprise_orchestration/__init__.py": "\"\"\"Claire enterprise orchestration foundation.\"\"\"\n",
  "src/claire/enterprise_orchestration/operator_role_registry.py": "from __future__ import annotations\nfrom typing import Dict, Any, List\n\nclass OperatorRoleRegistry:\n    def build_roles(self) -> Dict[str, Any]:\n        return {\"record_type\": \"operator_roles\", \"roles\": [\"owner\", \"operator\", \"reviewer\", \"observer\"], \"rule\": \"least privilege by default\"}\n",
  "src/claire/enterprise_orchestration/workspace_manifest.py": "from __future__ import annotations\nfrom typing import Dict, Any\n\nclass WorkspaceManifest:\n    def build_manifest(self, workspace_id: str, purpose: str) -> Dict[str, Any]:\n        return {\"record_type\": \"workspace_manifest\", \"workspace_id\": workspace_id, \"purpose\": purpose, \"isolation\": \"logical\"}\n",
  "src/claire/enterprise_orchestration/access_review_gate.py": "from __future__ import annotations\nfrom typing import Dict, Any\n\nclass AccessReviewGate:\n    def evaluate(self, reviewed: bool, owner_approved: bool) -> Dict[str, Any]:\n        return {\"status\": \"pass\" if reviewed and owner_approved else \"blocked\"}\n",
  "docs/enterprise_orchestration/v13_enterprise_orchestration_plan.md": "# v13 Enterprise Orchestration\n\nMulti-operator roles, workspace manifests, tenant boundaries, and access reviews.\n",
  "tests/enterprise_orchestration/test_enterprise_orchestration.py": "from claire.enterprise_orchestration.operator_role_registry import OperatorRoleRegistry\nfrom claire.enterprise_orchestration.access_review_gate import AccessReviewGate\n\ndef test_roles_include_owner():\n    assert \"owner\" in OperatorRoleRegistry().build_roles()[\"roles\"]\n\ndef test_access_blocks_unreviewed():\n    assert AccessReviewGate().evaluate(False, True)[\"status\"] == \"blocked\"\n"
}
VERSION_FILE = 'version_13_enterprise_orchestration.json'
LOCK_PAYLOAD = {
  "version": "13.5",
  "state": "enterprise_orchestration_foundation_installed",
  "phase": "enterprise_orchestration",
  "status": "installed_foundation_only",
  "safety_contract": {
    "does_not_modify_runtime": True,
    "does_not_modify_routing": True,
    "does_not_modify_scoring": True,
    "does_not_modify_core_run_output": True
  }
}

def stamp() -> str:
    return datetime.now(timezone.utc).strftime("%Y%m%d_%H%M%S")

def now_iso() -> str:
    return datetime.now(timezone.utc).isoformat()

def ensure_dir(path: Path, install: bool, actions: List[Dict[str, Any]]) -> None:
    existed = path.exists()
    if install:
        path.mkdir(parents=True, exist_ok=True)
    actions.append({"action": "ensure_dir", "path": str(path), "installed": install, "existed": existed})

def write_file(path: Path, content: str, install: bool, actions: List[Dict[str, Any]]) -> None:
    existed = path.exists()
    backup = None
    if install:
        path.parent.mkdir(parents=True, exist_ok=True)
        if existed:
            backup_path = path.with_name(path.name + ".bak_" + stamp())
            shutil.copy2(path, backup_path)
            backup = str(backup_path)
        path.write_text(content, encoding="utf-8", newline="\n")
    actions.append({"action": "write_file", "path": str(path), "installed": install, "existed": existed, "backup": backup})

def main() -> int:
    parser = argparse.ArgumentParser(description=INSTALLER_NAME)
    parser.add_argument("--target", default=str(DEFAULT_TARGET), help="Claire project root")
    parser.add_argument("--install", action="store_true", help="Apply install. Without this, dry run only.")
    args = parser.parse_args()

    root = Path(args.target).expanduser().resolve()
    install = bool(args.install)

    print(f"{INSTALLER_NAME} {INSTALLER_VERSION}")
    print(f"Target: {root}")
    print("Mode:", "INSTALL" if install else "DRY RUN")

    if not root.exists():
        print(f"[ERROR] Target does not exist: {root}")
        return 2
    if not (root / "src").exists():
        print("[ERROR] This does not look like Claire root. Missing src folder.")
        return 3

    actions: List[Dict[str, Any]] = []
    for folder in FOLDERS:
        ensure_dir(root / folder, install, actions)
    for rel, content in FILES.items():
        write_file(root / rel, content, install, actions)

    payload = dict(LOCK_PAYLOAD)
    payload["installed_at_utc"] = now_iso()
    write_file(root / VERSION_FILE, json.dumps(payload, indent=2, sort_keys=True), install, actions)

    report = {
        "installer": INSTALLER_NAME,
        "installer_version": INSTALLER_VERSION,
        "installed": install,
        "installed_at_utc": now_iso(),
        "target": str(root),
        "actions": actions,
        "version_file": VERSION_FILE,
        "safety_contract": payload.get("safety_contract", {}),
        "post_install_checks": [
            "Run the targeted pytest folder listed for this installer.",
            "Confirm new src/claire package folder exists.",
            "Confirm new data and docs folders exist.",
            "Commit after tests pass."
        ]
    }

    if install:
        report_path = root / ".claire_install" / "reports" / (VERSION_FILE.replace(".json", "") + "_" + stamp() + ".json")
        report_path.parent.mkdir(parents=True, exist_ok=True)
        report_path.write_text(json.dumps(report, indent=2, sort_keys=True), encoding="utf-8")
        print(f"[OK] Report written: {report_path}")
    else:
        print("[DRY RUN] No files changed. Re-run with --install to apply.")

    print(json.dumps({
        "installed": install,
        "target": str(root),
        "installer_version": INSTALLER_VERSION,
        "folders": len(FOLDERS),
        "files": len(FILES) + 1
    }, indent=2))
    return 0

if __name__ == "__main__":
    raise SystemExit(main())
