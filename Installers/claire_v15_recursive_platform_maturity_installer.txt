#!/usr/bin/env python3
from __future__ import annotations

import argparse
import json
import shutil
from datetime import datetime, timezone
from pathlib import Path
from typing import Any, Dict, List

INSTALLER_NAME = 'Claire v15 Recursive Platform Maturity Installer'
INSTALLER_VERSION = '15.0-15.5'
DEFAULT_TARGET = Path.home() / "OneDrive" / "Desktop" / "Claire"

FOLDERS = [
  "src/claire/recursive_platform_maturity",
  "tests/recursive_platform_maturity",
  "data/recursive_platform_maturity/self_ingestion_runs",
  "data/recursive_platform_maturity/system_transformation_maps",
  "data/recursive_platform_maturity/maturity_cycles",
  "data/recursive_platform_maturity/acquisition_ready_packages",
  "docs/recursive_platform_maturity",
  ".claire_install/reports"
]
FILES = {
  "src/claire/recursive_platform_maturity/__init__.py": "\"\"\"Claire recursive platform maturity foundation.\"\"\"\n",
  "src/claire/recursive_platform_maturity/self_ingestion_contract.py": "from __future__ import annotations\nfrom typing import Dict, Any, List\n\nclass SelfIngestionContract:\n    def build_contract(self, artifact_ids: List[str]) -> Dict[str, Any]:\n        return {\"record_type\": \"self_ingestion_contract\", \"artifact_ids\": artifact_ids, \"requires_human_review\": True, \"safe_mode\": True}\n",
  "src/claire/recursive_platform_maturity/system_transformation_map.py": "from __future__ import annotations\nfrom typing import Dict, Any, List\n\nclass SystemTransformationMap:\n    def build_map(self, source_system: str, improvement_targets: List[str]) -> Dict[str, Any]:\n        return {\"record_type\": \"system_transformation_map\", \"source_system\": source_system, \"improvement_targets\": improvement_targets}\n",
  "src/claire/recursive_platform_maturity/maturity_cycle_tracker.py": "from __future__ import annotations\nfrom typing import Dict, Any\n\nclass MaturityCycleTracker:\n    def record_cycle(self, cycle_id: str, result: str) -> Dict[str, Any]:\n        return {\"record_type\": \"maturity_cycle\", \"cycle_id\": cycle_id, \"result\": result, \"status\": \"recorded\"}\n",
  "docs/recursive_platform_maturity/v15_recursive_platform_maturity_plan.md": "# v15 Recursive Platform Maturity\n\nSelf-ingestion contracts, system transformation maps, maturity cycles, and acquisition-ready package foundations. Safe/review-gated only.\n",
  "tests/recursive_platform_maturity/test_recursive_platform_maturity.py": "from claire.recursive_platform_maturity.self_ingestion_contract import SelfIngestionContract\nfrom claire.recursive_platform_maturity.maturity_cycle_tracker import MaturityCycleTracker\n\ndef test_self_ingestion_safe_mode():\n    assert SelfIngestionContract().build_contract([\"a\"])[\"safe_mode\"] is True\n\ndef test_cycle_recorded():\n    assert MaturityCycleTracker().record_cycle(\"c1\", \"ok\")[\"status\"] == \"recorded\"\n"
}
VERSION_FILE = 'version_15_recursive_platform_maturity.json'
LOCK_PAYLOAD = {
  "version": "15.5",
  "state": "recursive_platform_maturity_foundation_installed",
  "phase": "recursive_platform_maturity",
  "status": "safe_review_gated_foundation_only",
  "safety_contract": {
    "does_not_modify_runtime": True,
    "does_not_modify_routing": True,
    "does_not_modify_scoring": True,
    "does_not_modify_core_run_output": True
  }
}

def stamp() -> str:
    return datetime.now(timezone.utc).strftime("%Y%m%d_%H%M%S")

def now_iso() -> str:
    return datetime.now(timezone.utc).isoformat()

def ensure_dir(path: Path, install: bool, actions: List[Dict[str, Any]]) -> None:
    existed = path.exists()
    if install:
        path.mkdir(parents=True, exist_ok=True)
    actions.append({"action": "ensure_dir", "path": str(path), "installed": install, "existed": existed})

def write_file(path: Path, content: str, install: bool, actions: List[Dict[str, Any]]) -> None:
    existed = path.exists()
    backup = None
    if install:
        path.parent.mkdir(parents=True, exist_ok=True)
        if existed:
            backup_path = path.with_name(path.name + ".bak_" + stamp())
            shutil.copy2(path, backup_path)
            backup = str(backup_path)
        path.write_text(content, encoding="utf-8", newline="\n")
    actions.append({"action": "write_file", "path": str(path), "installed": install, "existed": existed, "backup": backup})

def main() -> int:
    parser = argparse.ArgumentParser(description=INSTALLER_NAME)
    parser.add_argument("--target", default=str(DEFAULT_TARGET), help="Claire project root")
    parser.add_argument("--install", action="store_true", help="Apply install. Without this, dry run only.")
    args = parser.parse_args()

    root = Path(args.target).expanduser().resolve()
    install = bool(args.install)

    print(f"{INSTALLER_NAME} {INSTALLER_VERSION}")
    print(f"Target: {root}")
    print("Mode:", "INSTALL" if install else "DRY RUN")

    if not root.exists():
        print(f"[ERROR] Target does not exist: {root}")
        return 2
    if not (root / "src").exists():
        print("[ERROR] This does not look like Claire root. Missing src folder.")
        return 3

    actions: List[Dict[str, Any]] = []
    for folder in FOLDERS:
        ensure_dir(root / folder, install, actions)
    for rel, content in FILES.items():
        write_file(root / rel, content, install, actions)

    payload = dict(LOCK_PAYLOAD)
    payload["installed_at_utc"] = now_iso()
    write_file(root / VERSION_FILE, json.dumps(payload, indent=2, sort_keys=True), install, actions)

    report = {
        "installer": INSTALLER_NAME,
        "installer_version": INSTALLER_VERSION,
        "installed": install,
        "installed_at_utc": now_iso(),
        "target": str(root),
        "actions": actions,
        "version_file": VERSION_FILE,
        "safety_contract": payload.get("safety_contract", {}),
        "post_install_checks": [
            "Run the targeted pytest folder listed for this installer.",
            "Confirm new src/claire package folder exists.",
            "Confirm new data and docs folders exist.",
            "Commit after tests pass."
        ]
    }

    if install:
        report_path = root / ".claire_install" / "reports" / (VERSION_FILE.replace(".json", "") + "_" + stamp() + ".json")
        report_path.parent.mkdir(parents=True, exist_ok=True)
        report_path.write_text(json.dumps(report, indent=2, sort_keys=True), encoding="utf-8")
        print(f"[OK] Report written: {report_path}")
    else:
        print("[DRY RUN] No files changed. Re-run with --install to apply.")

    print(json.dumps({
        "installed": install,
        "target": str(root),
        "installer_version": INSTALLER_VERSION,
        "folders": len(FOLDERS),
        "files": len(FILES) + 1
    }, indent=2))
    return 0

if __name__ == "__main__":
    raise SystemExit(main())
