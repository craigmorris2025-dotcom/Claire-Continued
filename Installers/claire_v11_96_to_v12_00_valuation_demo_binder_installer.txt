#!/usr/bin/env python3
from __future__ import annotations

import argparse
import json
import shutil
from datetime import datetime, timezone
from pathlib import Path
from typing import Any, Dict, List

INSTALLER_NAME = 'Claire v11.96-v12.00 Valuation Demo Binder Installer'
INSTALLER_VERSION = '11.96-12.00'
DEFAULT_TARGET = Path.home() / "OneDrive" / "Desktop" / "Claire"

FOLDERS = [
  "src/claire/valuation_demo_binder",
  "tests/valuation_demo_binder",
  "data/valuation_demo_binder/demo_evidence",
  "data/valuation_demo_binder/valuation_memos",
  "data/valuation_demo_binder/screenshots_index",
  "data/valuation_demo_binder/benchmark_exhibits",
  "docs/valuation_demo_binder",
  ".claire_install/reports"
]
FILES = {
  "src/claire/valuation_demo_binder/__init__.py": "\"\"\"Claire valuation and demo binder.\"\"\"\n",
  "src/claire/valuation_demo_binder/valuation_memo_builder.py": "from __future__ import annotations\nfrom datetime import datetime, timezone\nfrom typing import Dict, Any, List\n\nclass ValuationMemoBuilder:\n    def build_memo(self, valuation_range: str, assumptions: List[str]) -> Dict[str, Any]:\n        return {\"record_type\": \"valuation_memo\", \"valuation_range\": valuation_range, \"assumptions\": assumptions, \"created_at_utc\": datetime.now(timezone.utc).isoformat(), \"disclaimer\": \"Preliminary strategic estimate, not financial advice.\"}\n",
  "src/claire/valuation_demo_binder/demo_evidence_binder.py": "from __future__ import annotations\nfrom typing import Dict, Any, List\n\nclass DemoEvidenceBinder:\n    def build_binder(self, artifacts: List[str]) -> Dict[str, Any]:\n        return {\"record_type\": \"demo_evidence_binder\", \"artifact_count\": len(artifacts), \"artifacts\": artifacts}\n",
  "src/claire/valuation_demo_binder/benchmark_exhibit_builder.py": "from __future__ import annotations\nfrom typing import Dict, Any\n\nclass BenchmarkExhibitBuilder:\n    def build_exhibit(self, benchmark_id: str, result: str) -> Dict[str, Any]:\n        return {\"record_type\": \"benchmark_exhibit\", \"benchmark_id\": benchmark_id, \"result\": result}\n",
  "docs/valuation_demo_binder/v11_96_to_v12_00_valuation_demo_binder_plan.md": "# v11.96-v12.00 Valuation/Demo Binder\n\nDemo evidence, valuation memo, screenshot index, benchmark exhibits, and binder structure.\n",
  "tests/valuation_demo_binder/test_valuation_demo_binder.py": "from claire.valuation_demo_binder.valuation_memo_builder import ValuationMemoBuilder\nfrom claire.valuation_demo_binder.demo_evidence_binder import DemoEvidenceBinder\n\ndef test_valuation_memo():\n    assert ValuationMemoBuilder().build_memo(\"$2M-$8M\", [\"demo\"])[\"record_type\"] == \"valuation_memo\"\n\ndef test_demo_binder_count():\n    assert DemoEvidenceBinder().build_binder([\"a\", \"b\"])[\"artifact_count\"] == 2\n"
}
VERSION_FILE = 'version_12_00_valuation_demo_binder.json'
LOCK_PAYLOAD = {
  "version": "12.00",
  "state": "valuation_demo_binder_ready",
  "phase": "valuation_demo_binder",
  "status": "installed_not_validated",
  "safety_contract": {
    "does_not_modify_runtime": True,
    "does_not_modify_routing": True,
    "does_not_modify_scoring": True,
    "does_not_modify_core_run_output": True
  }
}

def stamp() -> str:
    return datetime.now(timezone.utc).strftime("%Y%m%d_%H%M%S")

def now_iso() -> str:
    return datetime.now(timezone.utc).isoformat()

def ensure_dir(path: Path, install: bool, actions: List[Dict[str, Any]]) -> None:
    existed = path.exists()
    if install:
        path.mkdir(parents=True, exist_ok=True)
    actions.append({"action": "ensure_dir", "path": str(path), "installed": install, "existed": existed})

def write_file(path: Path, content: str, install: bool, actions: List[Dict[str, Any]]) -> None:
    existed = path.exists()
    backup = None
    if install:
        path.parent.mkdir(parents=True, exist_ok=True)
        if existed:
            backup_path = path.with_name(path.name + ".bak_" + stamp())
            shutil.copy2(path, backup_path)
            backup = str(backup_path)
        path.write_text(content, encoding="utf-8", newline="\n")
    actions.append({"action": "write_file", "path": str(path), "installed": install, "existed": existed, "backup": backup})

def main() -> int:
    parser = argparse.ArgumentParser(description=INSTALLER_NAME)
    parser.add_argument("--target", default=str(DEFAULT_TARGET), help="Claire project root")
    parser.add_argument("--install", action="store_true", help="Apply install. Without this, dry run only.")
    args = parser.parse_args()

    root = Path(args.target).expanduser().resolve()
    install = bool(args.install)

    print(f"{INSTALLER_NAME} {INSTALLER_VERSION}")
    print(f"Target: {root}")
    print("Mode:", "INSTALL" if install else "DRY RUN")

    if not root.exists():
        print(f"[ERROR] Target does not exist: {root}")
        return 2
    if not (root / "src").exists():
        print("[ERROR] This does not look like Claire root. Missing src folder.")
        return 3

    actions: List[Dict[str, Any]] = []
    for folder in FOLDERS:
        ensure_dir(root / folder, install, actions)
    for rel, content in FILES.items():
        write_file(root / rel, content, install, actions)

    payload = dict(LOCK_PAYLOAD)
    payload["installed_at_utc"] = now_iso()
    write_file(root / VERSION_FILE, json.dumps(payload, indent=2, sort_keys=True), install, actions)

    report = {
        "installer": INSTALLER_NAME,
        "installer_version": INSTALLER_VERSION,
        "installed": install,
        "installed_at_utc": now_iso(),
        "target": str(root),
        "actions": actions,
        "version_file": VERSION_FILE,
        "safety_contract": payload.get("safety_contract", {}),
        "post_install_checks": [
            "Run the targeted pytest folder listed for this installer.",
            "Confirm new src/claire package folder exists.",
            "Confirm new data and docs folders exist.",
            "Commit after tests pass."
        ]
    }

    if install:
        report_path = root / ".claire_install" / "reports" / (VERSION_FILE.replace(".json", "") + "_" + stamp() + ".json")
        report_path.parent.mkdir(parents=True, exist_ok=True)
        report_path.write_text(json.dumps(report, indent=2, sort_keys=True), encoding="utf-8")
        print(f"[OK] Report written: {report_path}")
    else:
        print("[DRY RUN] No files changed. Re-run with --install to apply.")

    print(json.dumps({
        "installed": install,
        "target": str(root),
        "installer_version": INSTALLER_VERSION,
        "folders": len(FOLDERS),
        "files": len(FILES) + 1
    }, indent=2))
    return 0

if __name__ == "__main__":
    raise SystemExit(main())
