def py_class_file(spec):
    role = spec.get("role", "Claire operational evidence module")
    class_name = spec.get("class_name", "ClaireEvidenceModule")
    writes_to = spec.get("writes_to", "data/operational_evidence")
    methods = spec.get("methods", [])
    lines = []
    lines.append("# " + role)
    lines.append("# Generated by Claire operational evidence infrastructure installer " + INSTALLER_VERSION)
    lines.append("")
    lines.append("import json")
    lines.append("import hashlib")
    lines.append("import logging")
    lines.append("from datetime import datetime")
    lines.append("from pathlib import Path")
    lines.append("")
    lines.append("logger = logging.getLogger(__name__)")
    lines.append("")
    lines.append("class " + class_name + ":")
    lines.append("    OUTPUT_DIR = Path(" + repr(writes_to) + ")")
    lines.append("    def __init__(self, output_dir=None):")
    lines.append("        self.output_dir = Path(output_dir) if output_dir else self.OUTPUT_DIR")
    lines.append("        self.logger = logging.getLogger(self.__class__.__name__)")
    lines.append("    def _artifact(self, method, **extra):")
    lines.append("        payload = {'status': 'success', 'component': self.__class__.__name__, 'method': method, 'generated_at': datetime.now().isoformat()}")
    lines.append("        payload.update(extra)")
    lines.append("        return payload")
    lines.append("    def _hash_payload(self, payload):")
    lines.append("        raw = json.dumps(payload, sort_keys=True, default=str).encode('utf-8')")
    lines.append("        return hashlib.sha256(raw).hexdigest()[:16]")
    lines.append("    def _save_json(self, name, payload):")
    lines.append("        self.output_dir.mkdir(parents=True, exist_ok=True)")
    lines.append("        path = self.output_dir / name")
    lines.append("        path.write_text(json.dumps(payload, indent=2, default=str), encoding='utf-8')")
    lines.append("        return str(path)")
    for method in methods:
        lines.append("    def " + method + "(self, *args, **kwargs):")
        if "hash" in method:
            lines.append("        payload = {'args': args, 'kwargs': kwargs}")
            lines.append("        return self._artifact(" + repr(method) + ", hash=self._hash_payload(payload))")
        elif any(x in method for x in ["validate", "check", "approve", "reject", "resolve"]):
            lines.append("        return self._artifact(" + repr(method) + ", valid=True, errors=[], warnings=[])")
        elif any(x in method for x in ["score"]):
            lines.append("        return self._artifact(" + repr(method) + ", score=0.0, level='unscored')")
        elif any(x in method for x in ["list", "query", "load", "get"]):
            lines.append("        return self._artifact(" + repr(method) + ", items=[])")
        else:
            lines.append("        return self._artifact(" + repr(method) + ")")
    if not methods:
        lines.append("    def run(self, *args, **kwargs):")
        lines.append("        return self._artifact('run')")
    return "\n".join(lines) + "\n"

    return "# " + spec.get("role", "Package init") + "\n__version__ = " + repr(INSTALLER_VERSION) + "\n__all__ = []\n"

def test_file(spec, rel_path):
    name = Path(rel_path).stem
    return "# " + spec.get("role", "Claire evidence operations smoke test") + "\n\ndef test_" + name + "_placeholder():\n    assert True\n"

def create_file_content(rel_path, spec):
    kind = spec.get("kind", Path(rel_path).suffix.lower().lstrip("."))
    if kind == "init":
        return init_file(spec)
    if kind == "py":
        return py_class_file(spec)
    if kind == "js":
        return js_file(spec)
    if kind == "test":
        return test_file(spec, rel_path)
    return "# Claire generated operational evidence file\n"

def preflight_check(target):
    checks = {
        "target_exists": target.is_dir(),
        "has_src": (target / "src").is_dir(),
        "has_frontend": (target / "src" / "frontend").is_dir(),
        "has_main": (target / "main.py").is_file(),
        "has_version_json": (target / "version.json").is_file(),
    }
    print("\n  Preflight checks:")
    ok = True
    for name, passed in checks.items():
        print(f"    [{'✓' if passed else '✗'}] {name}")
        ok = ok and passed
    return ok

def install_phase(phase_def, target, dry_run=True):
    print(f"\n  ── Phase {phase_def['phase']}: {phase_def['name']} ({phase_def['version']}) ──")
    result = {"phase": phase_def["phase"], "name": phase_def["name"], "version": phase_def["version"], "folders_created": [], "files_created": [], "files_skipped": [], "errors": []}
    for folder in phase_def.get("new_folders", []):
        path = target / folder
        if dry_run:
            print(f"    [DIR]  {folder} ({'EXISTS' if path.exists() else 'NEW'})")
            result["folders_created"].append(folder)
        else:
            try:
                path.mkdir(parents=True, exist_ok=True)
                if folder.startswith("data/"):
                    (path / ".gitkeep").touch(exist_ok=True)
                print(f"    [DIR]  ✓ {folder}")
                result["folders_created"].append(folder)
            except Exception as exc:
                print(f"    [DIR]  ✗ {folder}: {exc}")
                result["errors"].append(f"folder:{folder}:{exc}")
    for rel_path, spec in phase_def.get("new_files", {}).items():
        path = target / rel_path
        if path.exists():
            print(f"    [FILE] ⊘ {rel_path} (already exists, skipped)")
            result["files_skipped"].append(rel_path)
            continue
        if dry_run:
            print(f"    [FILE] {rel_path} ({spec.get('role', '')})")
            result["files_created"].append(rel_path)
        else:
            try:
                path.parent.mkdir(parents=True, exist_ok=True)
                path.write_text(create_file_content(rel_path, spec), encoding="utf-8")
                print(f"    [FILE] ✓ {rel_path}")
                result["files_created"].append(rel_path)
            except Exception as exc:
                print(f"    [FILE] ✗ {rel_path}: {exc}")
                result["errors"].append(f"file:{rel_path}:{exc}")
    print(f"    {'DRY RUN' if dry_run else 'INSTALLED'}: {len(result['folders_created'])} folders, {len(result['files_created'])} files, {len(result['files_skipped'])} skipped, {len(result['errors'])} errors")
    return result

def write_audit_log(target, results, dry_run):
    if dry_run:
        return
    log_dir = target / "data" / "update_logs"
    log_dir.mkdir(parents=True, exist_ok=True)
    stamp = datetime.now().strftime("%Y%m%d_%H%M%S")
    clean_version = INSTALLER_VERSION.replace(".", "_").replace("-", "_")
    path = log_dir / f"install_{stamp}_operational_evidence_{clean_version}_{INSTALLER_ID}.json"
    audit = {
        "installer_version": INSTALLER_VERSION,
        "installer_title": INSTALLER_TITLE,
        "installer_id": INSTALLER_ID,
        "installer_date": INSTALLER_DATE,
        "install_timestamp": datetime.now().isoformat(),
        "target_path": str(target),
        "phase_results": results,
        "total_files_created": sum(len(r["files_created"]) for r in results),
        "total_files_skipped": sum(len(r["files_skipped"]) for r in results),
        "total_errors": sum(len(r["errors"]) for r in results),
    }
    path.write_text(json.dumps(audit, indent=2), encoding="utf-8")
    print(f"\n  Audit log: {path}")

def show_summary():
    total_files = sum(len(p["new_files"]) for p in PHASES)
    total_folders = sum(len(p["new_folders"]) for p in PHASES)
    print(f"\n{INSTALLER_TITLE}")
    print(f"Claire v{INSTALLER_VERSION}: {len(PHASES)} phases, {total_files} files, {total_folders} folders\n")
    for p in PHASES:
        deps = ", ".join(str(d) for d in p["depends_on"]) if p["depends_on"] else "none"
        print(f"  Phase {p['phase']}: {p['name']} ({p['version']}) — {len(p['new_files'])} files, {len(p['new_folders'])} folders | depends on: {deps}")

def main():
    parser = argparse.ArgumentParser(description=INSTALLER_TITLE)
    parser.add_argument("--install", action="store_true", help="Actually install; default is dry run")
    parser.add_argument("--phase", type=int, choices=[p["phase"] for p in PHASES], help="Install only one phase")
    parser.add_argument("--target", type=str, default=DEFAULT_TARGET, help=f"Project root; default: {DEFAULT_TARGET}")
    parser.add_argument("--summary", action="store_true", help="Show summary")
    args = parser.parse_args()

    target = Path(args.target).resolve()
    dry_run = not args.install

    print("\n" + "═"*64)
    print(f"  {INSTALLER_TITLE}")
    print(f"  Target: {target}")
    print(f"  Mode: {'INSTALL' if args.install else 'DRY RUN'}")
    print("═"*64)

    if args.summary:
        show_summary()
        return

    if not preflight_check(target) and args.install:
        print("\n  ✗ Aborting install due to failed preflight checks.")
        sys.exit(1)

    phases = [p for p in PHASES if not args.phase or p["phase"] == args.phase]
    results = [install_phase(p, target, dry_run) for p in phases]
    write_audit_log(target, results, dry_run)

    print("\n" + "═"*64)
    print(f"  {'INSTALLATION COMPLETE' if args.install else 'DRY RUN COMPLETE'}")
    print(f"  {len(results)} phases | {sum(len(r['folders_created']) for r in results)} folders | {sum(len(r['files_created']) for r in results)} files | {sum(len(r['files_skipped']) for r in results)} skipped | {sum(len(r['errors']) for r in results)} errors")
    if dry_run:
