#!/usr/bin/env python3
from __future__ import annotations

import argparse
import json
import shutil
from datetime import datetime, timezone
from pathlib import Path
from typing import Any, Dict, List

INSTALLER_NAME = 'Claire v11.91-v11.95 IP Transfer Package Installer'
INSTALLER_VERSION = '11.91-11.95'
DEFAULT_TARGET = Path.home() / "OneDrive" / "Desktop" / "Claire"

FOLDERS = [
  "src/claire/ip_transfer_package",
  "tests/ip_transfer_package",
  "data/ip_transfer_package/repo_inventory",
  "data/ip_transfer_package/ownership_records",
  "data/ip_transfer_package/transfer_checklists",
  "data/ip_transfer_package/license_notes",
  "docs/ip_transfer_package",
  ".claire_install/reports"
]
FILES = {
  "src/claire/ip_transfer_package/__init__.py": "\"\"\"Claire IP transfer package prep.\"\"\"\n",
  "src/claire/ip_transfer_package/repo_inventory_builder.py": "from __future__ import annotations\nfrom pathlib import Path\nfrom typing import Dict, Any\n\nclass RepoInventoryBuilder:\n    def __init__(self, root: str | Path = \".\") -> None:\n        self.root = Path(root)\n    def build_inventory(self) -> Dict[str, Any]:\n        files = [str(p.relative_to(self.root)) for p in self.root.rglob(\"*\") if p.is_file() and \".git\" not in p.parts]\n        return {\"record_type\": \"repo_inventory\", \"file_count\": len(files), \"sample\": files[:100]}\n",
  "src/claire/ip_transfer_package/ownership_record_builder.py": "from __future__ import annotations\nfrom datetime import datetime, timezone\nfrom typing import Dict, Any\n\nclass OwnershipRecordBuilder:\n    def build_record(self, owner: str = \"Craig Morris\", project: str = \"Claire Syntalion\") -> Dict[str, Any]:\n        return {\"record_type\": \"ownership_record\", \"owner\": owner, \"project\": project, \"created_at_utc\": datetime.now(timezone.utc).isoformat(), \"note\": \"Draft record only; not legal advice.\"}\n",
  "src/claire/ip_transfer_package/transfer_checklist.py": "from __future__ import annotations\nfrom typing import Dict, Any\n\nclass TransferChecklist:\n    REQUIRED = [\"repo_inventory\", \"version_lock\", \"readme\", \"license_note\", \"demo_artifacts\", \"pytest_proof\"]\n    def evaluate(self, checks: Dict[str, bool]) -> Dict[str, Any]:\n        missing = [x for x in self.REQUIRED if not checks.get(x, False)]\n        return {\"status\": \"ready\" if not missing else \"incomplete\", \"missing\": missing}\n",
  "docs/ip_transfer_package/v11_91_to_v11_95_ip_transfer_plan.md": "# v11.91-v11.95 IP Transfer Package Prep\n\nDraft ownership records, repo inventory, transfer checklist, and license notes. Not legal advice.\n",
  "tests/ip_transfer_package/test_ip_transfer_package.py": "from claire.ip_transfer_package.transfer_checklist import TransferChecklist\nfrom claire.ip_transfer_package.ownership_record_builder import OwnershipRecordBuilder\n\ndef test_transfer_checklist_incomplete():\n    assert TransferChecklist().evaluate({})[\"status\"] == \"incomplete\"\n\ndef test_ownership_record():\n    assert OwnershipRecordBuilder().build_record()[\"owner\"] == \"Craig Morris\"\n"
}
VERSION_FILE = 'version_11_95_ip_transfer_package.json'
LOCK_PAYLOAD = {
  "version": "11.95",
  "state": "ip_transfer_package_installed",
  "phase": "transfer_ip_prep",
  "status": "draft_not_legal_final",
  "safety_contract": {
    "does_not_modify_runtime": True,
    "does_not_modify_routing": True,
    "does_not_modify_scoring": True,
    "does_not_modify_core_run_output": True
  }
}

def stamp() -> str:
    return datetime.now(timezone.utc).strftime("%Y%m%d_%H%M%S")

def now_iso() -> str:
    return datetime.now(timezone.utc).isoformat()

def ensure_dir(path: Path, install: bool, actions: List[Dict[str, Any]]) -> None:
    existed = path.exists()
    if install:
        path.mkdir(parents=True, exist_ok=True)
    actions.append({"action": "ensure_dir", "path": str(path), "installed": install, "existed": existed})

def write_file(path: Path, content: str, install: bool, actions: List[Dict[str, Any]]) -> None:
    existed = path.exists()
    backup = None
    if install:
        path.parent.mkdir(parents=True, exist_ok=True)
        if existed:
            backup_path = path.with_name(path.name + ".bak_" + stamp())
            shutil.copy2(path, backup_path)
            backup = str(backup_path)
        path.write_text(content, encoding="utf-8", newline="\n")
    actions.append({"action": "write_file", "path": str(path), "installed": install, "existed": existed, "backup": backup})

def main() -> int:
    parser = argparse.ArgumentParser(description=INSTALLER_NAME)
    parser.add_argument("--target", default=str(DEFAULT_TARGET), help="Claire project root")
    parser.add_argument("--install", action="store_true", help="Apply install. Without this, dry run only.")
    args = parser.parse_args()

    root = Path(args.target).expanduser().resolve()
    install = bool(args.install)

    print(f"{INSTALLER_NAME} {INSTALLER_VERSION}")
    print(f"Target: {root}")
    print("Mode:", "INSTALL" if install else "DRY RUN")

    if not root.exists():
        print(f"[ERROR] Target does not exist: {root}")
        return 2
    if not (root / "src").exists():
        print("[ERROR] This does not look like Claire root. Missing src folder.")
        return 3

    actions: List[Dict[str, Any]] = []
    for folder in FOLDERS:
        ensure_dir(root / folder, install, actions)
    for rel, content in FILES.items():
        write_file(root / rel, content, install, actions)

    payload = dict(LOCK_PAYLOAD)
    payload["installed_at_utc"] = now_iso()
    write_file(root / VERSION_FILE, json.dumps(payload, indent=2, sort_keys=True), install, actions)

    report = {
        "installer": INSTALLER_NAME,
        "installer_version": INSTALLER_VERSION,
        "installed": install,
        "installed_at_utc": now_iso(),
        "target": str(root),
        "actions": actions,
        "version_file": VERSION_FILE,
        "safety_contract": payload.get("safety_contract", {}),
        "post_install_checks": [
            "Run the targeted pytest folder listed for this installer.",
            "Confirm new src/claire package folder exists.",
            "Confirm new data and docs folders exist.",
            "Commit after tests pass."
        ]
    }

    if install:
        report_path = root / ".claire_install" / "reports" / (VERSION_FILE.replace(".json", "") + "_" + stamp() + ".json")
        report_path.parent.mkdir(parents=True, exist_ok=True)
        report_path.write_text(json.dumps(report, indent=2, sort_keys=True), encoding="utf-8")
        print(f"[OK] Report written: {report_path}")
    else:
        print("[DRY RUN] No files changed. Re-run with --install to apply.")

    print(json.dumps({
        "installed": install,
        "target": str(root),
        "installer_version": INSTALLER_VERSION,
        "folders": len(FOLDERS),
        "files": len(FILES) + 1
    }, indent=2))
    return 0

if __name__ == "__main__":
    raise SystemExit(main())
