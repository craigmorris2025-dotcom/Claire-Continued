import json
from datetime import datetime, timezone
from pathlib import Path
from typing import Dict, List

                "known_placeholder",
                "baseline_runner_guard",
                "expected_failures",

    status = "pass" if not active_placeholder_tests and baseline_guarded is not False else "needs_attention"

        "status": status,
        "created_at_utc": datetime.now(timezone.utc).isoformat(),
        "active_placeholder_tests": active_placeholder_tests,

        "baseline_runner_exists": baseline.exists(),

## Problem

Full pytest is repeatedly being destabilized by older placeholder tests and baseline assumptions that no longer match the current Claire maturity state.

Typical failure shape:

- `NotImplementedError: Import target class`
- old stage tests never bound to real runtime classes
- baseline checks expecting zero records/failures while the system now has installed proof/governance infrastructure

## Rule

A test may be one of three things:

1. Active proof test — must import a real class and pass.
2. Bound stage contract test — must point to a real stage/runtime class.
3. Future placeholder — must be underscore-disabled, documented, or marked until the target class exists.

A placeholder test must not break full pytest.

def now_iso() -> str:
    return datetime.now(timezone.utc).isoformat()


def stamp() -> str:
    return datetime.now(timezone.utc).strftime("%Y%m%d_%H%M%S")


def write_file(path: Path, content: str, install: bool, actions: List[Dict[str, Any]], overwrite: bool) -> None:
    existed = path.exists()
    backup = None

    if install:
        path.parent.mkdir(parents=True, exist_ok=True)
        if existed and not overwrite:
            actions.append({"action": "skip_existing_file", "path": str(path), "installed": False, "existed": existed})
            return
        if existed and overwrite:
            backup_path = path.with_name(path.name + ".bak_" + stamp())
            shutil.copy2(path, backup_path)
            backup = str(backup_path)
        path.write_text(content, encoding="utf-8", newline="\n")

    actions.append({"action": "write_file", "path": str(path), "installed": install, "existed": existed, "backup": backup})


def version_payload() -> Dict[str, Any]:
    return {
        "version": "11.50.1",
        "state": "test_consistency_guard_installed",
        "installed_at_utc": now_iso(),
        "phase": "operational_proof_plateau_consistency",
        "status": "installed_not_validated",
        "purpose": "Prevent recurring pytest instability from unbound placeholder tests and stale baseline assumptions.",
        "safety_contract": {
            "does_not_modify_runtime": True,
            "does_not_modify_routing": True,
            "does_not_modify_scoring": True,
            "does_not_modify_dashboard": True,
            "does_not_modify_core_run_output": True,
            "test_layer_only": True,
        },
        "validation_commands": [
            "pytest tests/consistency -v",
            "pytest tests/operational_proof -v",
            "python tools/pytest_consistency_audit.py",
            "pytest",
        ],
    }


def main() -> int:
    parser = argparse.ArgumentParser(description=INSTALLER_NAME)
    parser.add_argument("--target", default=str(DEFAULT_TARGET), help="Claire project root")
    parser.add_argument("--install", action="store_true", help="Apply install. Without this, dry run only.")
    parser.add_argument("--overwrite", action="store_true", help="Overwrite files installed by this package if they already exist.")
    args = parser.parse_args()

    root = Path(args.target).expanduser().resolve()
    install = bool(args.install)

    print(f"{INSTALLER_NAME} {INSTALLER_VERSION}")
    print(f"Target: {root}")
    print("Mode:", "INSTALL" if install else "DRY RUN")

    if not root.exists():
        print(f"[ERROR] Target does not exist: {root}")
        return 2

    if not (root / "src").exists() or not (root / "tests").exists():
        print("[ERROR] This does not look like the Claire project root. Missing src or tests folder.")
        return 3

    actions: List[Dict[str, Any]] = []

    for folder in FOLDERS:
        path = root / folder
        existed = path.exists()
        if install:
            path.mkdir(parents=True, exist_ok=True)
        actions.append({"action": "ensure_dir", "path": str(path), "installed": install, "existed": existed})

    for rel, content in FILES.items():
        write_file(root / rel, content, install, actions, overwrite=args.overwrite)

    write_file(
        root / "version_11_50_1_test_consistency_guard.json",
        json.dumps(version_payload(), indent=2, sort_keys=True),
        install,
        actions,
        overwrite=True,
    )

    report = {
        "installer": INSTALLER_NAME,
        "installer_version": INSTALLER_VERSION,
        "installed": install,
        "installed_at_utc": now_iso(),
        "target": str(root),
        "actions": actions,
        "post_install_checks": version_payload()["validation_commands"],
        "safety_contract": version_payload()["safety_contract"],
    }

    if install:
        report_path = root / ".claire_install" / "reports" / ("v11_50_1_test_consistency_guard_" + stamp() + ".json")
        report_path.parent.mkdir(parents=True, exist_ok=True)
        report_path.write_text(json.dumps(report, indent=2, sort_keys=True), encoding="utf-8")
        print(f"[OK] Report written: {report_path}")
    else:
        print("[DRY RUN] No files changed. Re-run with --install to apply.")

    print(json.dumps({
        "installed": install,
        "target": str(root),
        "version": INSTALLER_VERSION,
        "folders": len(FOLDERS),
        "files": len(FILES) + 1,
    }, indent=2))

    return 0


if __name__ == "__main__":
    raise SystemExit(main())
