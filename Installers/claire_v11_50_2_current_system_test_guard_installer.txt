from __future__ import annotations

import argparse
import json
from datetime import datetime, timezone
from pathlib import Path
from typing import Dict, List


def is_placeholder_stage_test(path: Path) -> bool:
    text = path.read_text(encoding="utf-8", errors="ignore")
    return "NotImplementedError" in text and "Import target class" in text


def find_active_placeholders(root: Path) -> List[Path]:
    found: List[Path] = []
    tests_root = root / "tests"
    if not tests_root.exists():
        return found

    for path in tests_root.rglob("test_*.py"):
        if "consistency" in path.parts:
            continue
        if is_placeholder_stage_test(path):
            found.append(path)

    return sorted(found)


def quarantine_name(path: Path) -> Path:
    if path.name.startswith("_"):
        return path
    candidate = path.with_name("_" + path.stem + "_placeholder" + path.suffix)
    if not candidate.exists():
        return candidate

    timestamp = datetime.now(timezone.utc).strftime("%Y%m%d_%H%M%S")
    return path.with_name("_" + path.stem + "_placeholder_" + timestamp + path.suffix)


def audit(root: Path) -> Dict[str, object]:
    active = find_active_placeholders(root)
    underscored = []

    for path in (root / "tests").rglob("_test*.py"):
        if is_placeholder_stage_test(path):
            underscored.append(path)

    return {
        "record_type": "current_system_pytest_guard_audit",
        "status": "pass" if not active else "needs_quarantine",
        "created_at_utc": datetime.now(timezone.utc).isoformat(),
        "active_placeholder_tests": [str(p.relative_to(root)) for p in active],
        "quarantined_placeholder_tests": [str(p.relative_to(root)) for p in sorted(underscored)],
        "rule": "Old Import target class placeholders must be quarantined until bound to real current-system classes.",

    out = root / "data" / "testing" / "consistency_audits" / (

    manifest.write_text(json.dumps(payload, indent=2, sort_keys=True), encoding="utf-8")

def now_iso() -> str:
    return datetime.now(timezone.utc).isoformat()


def stamp() -> str:
    return datetime.now(timezone.utc).strftime("%Y%m%d_%H%M%S")


def version_payload() -> Dict[str, Any]:
    return {
        "version": INSTALLER_VERSION,
        "state": "current_system_test_guard_installed",
        "installed_at_utc": now_iso(),
        "phase": "operational_proof_plateau_consistency",
        "status": "installed_not_validated",
        "purpose": "Quarantine old unbound stage placeholder tests and stabilize pytest against current Claire system state.",
        "safety_contract": {
            "does_not_modify_runtime": True,
            "does_not_modify_routing": True,
            "does_not_modify_scoring": True,
            "does_not_modify_dashboard": True,
            "does_not_modify_launcher": True,
            "does_not_modify_core_run_output": True,
            "test_layer_only": True,
            "renames_placeholder_tests_only_when_requested": True,
        },
        "validation_commands": [
            "pytest tests/consistency -v",
            "pytest tests/operational_proof -v",
            "python tools/pytest_current_system_guard.py --audit",

def write_file(path: Path, content: str, install: bool, actions: List[Dict[str, Any]], overwrite: bool) -> None:
    existed = path.exists()
    backup = None

    if install:
        path.parent.mkdir(parents=True, exist_ok=True)
        if existed and not overwrite:
            actions.append({
                "action": "skip_existing_file",
                "path": str(path),
                "installed": False,
                "existed": existed,
                "backup": None,
            })
            return

        if existed and overwrite:
            backup_path = path.with_name(path.name + ".bak_" + stamp())
            shutil.copy2(path, backup_path)
            backup = str(backup_path)

        path.write_text(content, encoding="utf-8", newline="\n")

    actions.append({
        "action": "write_file",
        "path": str(path),
        "installed": install,
        "existed": existed,
        "backup": backup,
    })


def is_placeholder_stage_test(path: Path) -> bool:
    text = path.read_text(encoding="utf-8", errors="ignore")
    return "NotImplementedError" in text and "Import target class" in text


def find_active_placeholders(root: Path) -> List[Path]:
    found: List[Path] = []
    for path in (root / "tests").rglob("test_*.py"):
        if "consistency" in path.parts:
            continue
        if is_placeholder_stage_test(path):
            found.append(path)
    return sorted(found)


def quarantine_name(path: Path) -> Path:
    candidate = path.with_name("_" + path.stem + "_placeholder" + path.suffix)
    if not candidate.exists():
        return candidate
    return path.with_name("_" + path.stem + "_placeholder_" + stamp() + path.suffix)


def quarantine_placeholders(root: Path, install: bool, actions: List[Dict[str, Any]]) -> List[Dict[str, str]]:
    moves: List[Dict[str, str]] = []
    active = find_active_placeholders(root)

    for src in active:
        dest = quarantine_name(src)
        if install:
            src.rename(dest)
        move = {
            "from": str(src.relative_to(root)),
            "to": str(dest.relative_to(root)),
            "installed": str(install),
            "reason": "active placeholder contained NotImplementedError('Import target class')",
        }
        moves.append(move)
        actions.append({"action": "quarantine_placeholder", **move})

    if install:
        manifest = root / "data" / "testing" / "quarantined_placeholders" / ("installer_quarantine_" + stamp() + ".json")
        manifest.parent.mkdir(parents=True, exist_ok=True)
        manifest.write_text(json.dumps({
            "record_type": "installer_placeholder_quarantine",
            "created_at_utc": now_iso(),
            "move_count": len(moves),
            "moves": moves,
        }, indent=2, sort_keys=True), encoding="utf-8")

    return moves


def main() -> int:
    parser = argparse.ArgumentParser(description=INSTALLER_NAME)
    parser.add_argument("--target", default=str(DEFAULT_TARGET), help="Claire project root")
    parser.add_argument("--install", action="store_true", help="Apply install. Without this, dry run only.")
    parser.add_argument("--overwrite", action="store_true", help="Overwrite this package's guard files if already present.")
    parser.add_argument("--quarantine-placeholders", action="store_true", help="Rename active Import target class placeholder tests so pytest ignores them.")
    args = parser.parse_args()

    root = Path(args.target).expanduser().resolve()
    install = bool(args.install)

    print(f"{INSTALLER_NAME} {INSTALLER_VERSION}")
    print(f"Target: {root}")
    print("Mode:", "INSTALL" if install else "DRY RUN")

    if not root.exists():
        print(f"[ERROR] Target does not exist: {root}")
        return 2

    if not (root / "src").exists() or not (root / "tests").exists():
        print("[ERROR] This does not look like the Claire project root. Missing src or tests folder.")
        return 3

    actions: List[Dict[str, Any]] = []

    for folder in FOLDERS:
        path = root / folder
        existed = path.exists()
        if install:
            path.mkdir(parents=True, exist_ok=True)
        actions.append({
            "action": "ensure_dir",
            "path": str(path),
            "installed": install,
            "existed": existed,
        })

    for rel, content in FILES.items():
        write_file(root / rel, content, install, actions, overwrite=args.overwrite)

    write_file(
        root / "version_11_50_2_current_system_test_guard.json",
        json.dumps(version_payload(), indent=2, sort_keys=True),
        install,
        actions,
        overwrite=True,
    )

    active_before = [str(p.relative_to(root)) for p in find_active_placeholders(root)]
    moves: List[Dict[str, str]] = []

    if args.quarantine_placeholders:
        moves = quarantine_placeholders(root, install, actions)

    active_after = [str(p.relative_to(root)) for p in find_active_placeholders(root)] if install else active_before

    report = {
        "installer": INSTALLER_NAME,
        "installer_version": INSTALLER_VERSION,
        "installed": install,
        "installed_at_utc": now_iso(),
        "target": str(root),
        "active_placeholders_before": active_before,
        "active_placeholders_after": active_after,
        "quarantine_requested": bool(args.quarantine_placeholders),
        "quarantine_moves": moves,
        "actions": actions,
        "post_install_checks": version_payload()["validation_commands"],
        "safety_contract": version_payload()["safety_contract"],
    }

    if install:
        report_path = root / ".claire_install" / "reports" / ("v11_50_2_current_system_test_guard_" + stamp() + ".json")
        report_path.parent.mkdir(parents=True, exist_ok=True)
        report_path.write_text(json.dumps(report, indent=2, sort_keys=True), encoding="utf-8")
        print(f"[OK] Report written: {report_path}")
    else:
        print("[DRY RUN] No files changed. Re-run with --install to apply.")

    print(json.dumps({
        "installed": install,
        "target": str(root),
        "version": INSTALLER_VERSION,
        "folders": len(FOLDERS),
        "files": len(FILES) + 1,
        "active_placeholders_before": active_before,
        "quarantine_requested": bool(args.quarantine_placeholders),
        "quarantine_moves": moves,
        "active_placeholders_after": active_after,
    }, indent=2, sort_keys=True))

if __name__ == "__main__":
    raise SystemExit(main())
