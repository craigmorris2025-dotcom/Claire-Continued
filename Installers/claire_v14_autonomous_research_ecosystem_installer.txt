#!/usr/bin/env python3
from __future__ import annotations

import argparse
import json
import shutil
from datetime import datetime, timezone
from pathlib import Path
from typing import Any, Dict, List

INSTALLER_NAME = 'Claire v14 Autonomous Research Ecosystem Installer'
INSTALLER_VERSION = '14.0-14.5'
DEFAULT_TARGET = Path.home() / "OneDrive" / "Desktop" / "Claire"

FOLDERS = [
  "src/claire/autonomous_research_ecosystem",
  "tests/autonomous_research_ecosystem",
  "data/autonomous_research_ecosystem/research_tasks",
  "data/autonomous_research_ecosystem/source_policies",
  "data/autonomous_research_ecosystem/synthesis_reviews",
  "data/autonomous_research_ecosystem/research_audits",
  "docs/autonomous_research_ecosystem",
  ".claire_install/reports"
]
FILES = {
  "src/claire/autonomous_research_ecosystem/__init__.py": "\"\"\"Claire autonomous research ecosystem foundation.\"\"\"\n",
  "src/claire/autonomous_research_ecosystem/research_task_contract.py": "from __future__ import annotations\nfrom typing import Dict, Any, List\n\nclass ResearchTaskContract:\n    def build_contract(self, task_id: str, question: str, allowed_sources: List[str]) -> Dict[str, Any]:\n        return {\"record_type\": \"research_task_contract\", \"task_id\": task_id, \"question\": question, \"allowed_sources\": allowed_sources, \"requires_review\": True}\n",
  "src/claire/autonomous_research_ecosystem/source_policy_gate.py": "from __future__ import annotations\nfrom typing import Dict, Any, List\n\nclass SourcePolicyGate:\n    def evaluate(self, source: str, allowlist: List[str]) -> Dict[str, Any]:\n        return {\"status\": \"allowed\" if source in allowlist else \"blocked\", \"source\": source}\n",
  "src/claire/autonomous_research_ecosystem/synthesis_review_queue.py": "from __future__ import annotations\nfrom typing import Dict, Any\n\nclass SynthesisReviewQueue:\n    def queue_item(self, item_id: str, summary: str) -> Dict[str, Any]:\n        return {\"record_type\": \"synthesis_review_item\", \"item_id\": item_id, \"summary\": summary, \"status\": \"queued_for_review\"}\n",
  "docs/autonomous_research_ecosystem/v14_autonomous_research_ecosystem_plan.md": "# v14 Autonomous Research Ecosystem\n\nResearch task contracts, source policy gates, synthesis review queues, and audits. Review-gated only.\n",
  "tests/autonomous_research_ecosystem/test_autonomous_research_ecosystem.py": "from claire.autonomous_research_ecosystem.source_policy_gate import SourcePolicyGate\nfrom claire.autonomous_research_ecosystem.synthesis_review_queue import SynthesisReviewQueue\n\ndef test_source_policy_blocks_unknown():\n    assert SourcePolicyGate().evaluate(\"x\", [])[\"status\"] == \"blocked\"\n\ndef test_review_queue():\n    assert SynthesisReviewQueue().queue_item(\"i1\", \"summary\")[\"status\"] == \"queued_for_review\"\n"
}
VERSION_FILE = 'version_14_autonomous_research_ecosystem.json'
LOCK_PAYLOAD = {
  "version": "14.5",
  "state": "autonomous_research_ecosystem_foundation_installed",
  "phase": "autonomous_research_ecosystem",
  "status": "review_gated_foundation_only",
  "safety_contract": {
    "does_not_modify_runtime": True,
    "does_not_modify_routing": True,
    "does_not_modify_scoring": True,
    "does_not_modify_core_run_output": True
  }
}

def stamp() -> str:
    return datetime.now(timezone.utc).strftime("%Y%m%d_%H%M%S")

def now_iso() -> str:
    return datetime.now(timezone.utc).isoformat()

def ensure_dir(path: Path, install: bool, actions: List[Dict[str, Any]]) -> None:
    existed = path.exists()
    if install:
        path.mkdir(parents=True, exist_ok=True)
    actions.append({"action": "ensure_dir", "path": str(path), "installed": install, "existed": existed})

def write_file(path: Path, content: str, install: bool, actions: List[Dict[str, Any]]) -> None:
    existed = path.exists()
    backup = None
    if install:
        path.parent.mkdir(parents=True, exist_ok=True)
        if existed:
            backup_path = path.with_name(path.name + ".bak_" + stamp())
            shutil.copy2(path, backup_path)
            backup = str(backup_path)
        path.write_text(content, encoding="utf-8", newline="\n")
    actions.append({"action": "write_file", "path": str(path), "installed": install, "existed": existed, "backup": backup})

def main() -> int:
    parser = argparse.ArgumentParser(description=INSTALLER_NAME)
    parser.add_argument("--target", default=str(DEFAULT_TARGET), help="Claire project root")
    parser.add_argument("--install", action="store_true", help="Apply install. Without this, dry run only.")
    args = parser.parse_args()

    root = Path(args.target).expanduser().resolve()
    install = bool(args.install)

    print(f"{INSTALLER_NAME} {INSTALLER_VERSION}")
    print(f"Target: {root}")
    print("Mode:", "INSTALL" if install else "DRY RUN")

    if not root.exists():
        print(f"[ERROR] Target does not exist: {root}")
        return 2
    if not (root / "src").exists():
        print("[ERROR] This does not look like Claire root. Missing src folder.")
        return 3

    actions: List[Dict[str, Any]] = []
    for folder in FOLDERS:
        ensure_dir(root / folder, install, actions)
    for rel, content in FILES.items():
        write_file(root / rel, content, install, actions)

    payload = dict(LOCK_PAYLOAD)
    payload["installed_at_utc"] = now_iso()
    write_file(root / VERSION_FILE, json.dumps(payload, indent=2, sort_keys=True), install, actions)

    report = {
        "installer": INSTALLER_NAME,
        "installer_version": INSTALLER_VERSION,
        "installed": install,
        "installed_at_utc": now_iso(),
        "target": str(root),
        "actions": actions,
        "version_file": VERSION_FILE,
        "safety_contract": payload.get("safety_contract", {}),
        "post_install_checks": [
            "Run the targeted pytest folder listed for this installer.",
            "Confirm new src/claire package folder exists.",
            "Confirm new data and docs folders exist.",
            "Commit after tests pass."
        ]
    }

    if install:
        report_path = root / ".claire_install" / "reports" / (VERSION_FILE.replace(".json", "") + "_" + stamp() + ".json")
        report_path.parent.mkdir(parents=True, exist_ok=True)
        report_path.write_text(json.dumps(report, indent=2, sort_keys=True), encoding="utf-8")
        print(f"[OK] Report written: {report_path}")
    else:
        print("[DRY RUN] No files changed. Re-run with --install to apply.")

    print(json.dumps({
        "installed": install,
        "target": str(root),
        "installer_version": INSTALLER_VERSION,
        "folders": len(FOLDERS),
        "files": len(FILES) + 1
    }, indent=2))
    return 0

if __name__ == "__main__":
    raise SystemExit(main())
